Add the environment variables

export HADOOP_HOME=/usr/local/hadoop
export HADOOP_INSTALL=$HADOOP_HOME
export HADOOP_MAPRED_HOME=$HADOOP_HOME
export HADOOP_COMMON_HOME=$HADOOP_HOME
export HADOOP_CONF_DIR=/usr/local/hadoop/etc/hadoop
export HADOOP_HDFS_HOME=$HADOOP_HOME
export YARN_HOME=$HADOOP_HOME
export HADOOP_COMMON_LIB_NATIVE_DIR=$HADOOP_HOME/lib/native
export PATH=$PATH:$HADOOP_HOME/sbin:$HADOOP_HOME/bin

Hadoop Configurations

$HADOOP_CONF_DIR/yarn-site.xml

    <configuration>
    <! — Site specific YARN configuration properties →
    <property>
      <name>yarn.nodemanager.aux-services</name>
      <value>mapreduce_shuffle</value>
    </property> 
    <property>
      <name>yarn.nodemanager.aux-services.mapreduce.shuffle.class</name>
      <value>org.apache.hadoop.mapred.ShuffleHandler</value>
    </property>
    <property>
      <name>yarn.resourcemanager.hostname</name>
      <value>namenode_public_dns</value>
    </property>
    </configuration>

$HADOOP_CONF_DIR/hadoop-env.sh
 export JAVA_HOME=/usr

Create a Oozie database and start the ooze service and check for the status of the ooze services

ooziedb.sh create -sqlfile oozie.sql -run
oozied.sh start
ozie admin --oozie http://localhost:/oozie -status


Compile the java programs (map reduce)

javac -classpath $HADOOP_HOME/share/hadoop/common/hadoop-common-2.6.0.jar:$HADOOP_HOME/share/hadoop/mapreduce/hadoop-mapreduce-client-core-2.6.0.jar:$HADOOP_HOME/share/hadoop/common/lib/commons-cli-1.2.jar -d ./ *.java





Create the jar file, map reduce and input directory on HDFS

jar -cvf bigdataoozie.jar -C classfiles/
hdfs dfs -mkdir input

Copy xml into ap reduce and jar file into ~/map-reduce/lib and make sure that the flight data is copied into HDFS and copy the map reduce to HDFS

hdfs dfs -put ~/map-reduce map-reduce

Run the ooze job

oozie job -oozie http://localhost:/oozie -config ~/map-reduce/job.properties -run
